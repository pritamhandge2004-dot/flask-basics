# Exercise 3.3: Create a grades page
@app.route('/grades')
def grades():
    grades_data = {
        'Python Programming': 'A+',
        'Web Development': 'A',
        'Database Systems': 'B+',
        # ... more subjects ...
    }
    return render_template('grades.html',
        grades=grades_data,
        average_grade=8.5,
        student_name="Pritam")
